import React, { useState, useEffect, useRef } from 'react';
import ReactMarkdown from 'react-markdown';
import remarkGfm from 'remark-gfm';
import rehypeHighlight from 'rehype-highlight';
import { X, FileText, CheckCircle, AlertCircle, Loader, Type } from 'lucide-react';
import { exportPDF } from '../utils/exportPDF';

interface BatchConverterProps {
  onClose: () => void;
}

interface FileStatus {
  file: File;
  status: 'pending' | 'converting' | 'success' | 'error';
  errorMessage?: string;
}

const BatchConverter: React.FC<BatchConverterProps> = ({ onClose }) => {
  const [fileQueue, setFileQueue] = useState<FileStatus[]>([]);
  const [isConverting, setIsConverting] = useState(false);
  const [currentIndex, setCurrentIndex] = useState<number>(-1);
  const [currentMarkdown, setCurrentMarkdown] = useState<string>('');
  const [cancelRequested, setCancelRequested] = useState(false);
  const fileInputRef = useRef<HTMLInputElement>(null);

  const handleFilesSelected = (e: React.ChangeEvent<HTMLInputElement>) => {
    if (e.target.files) {
      const selectedFiles = Array.from(e.target.files).filter(f => f.name.toLowerCase().endsWith('.md'));
      
      if (selectedFiles.length > 20) {
        alert('You can only upload up to 20 files at once.');
        return;
      }
      
      if (selectedFiles.length === 0) {
        alert('Please select valid .md files.');
        return;
      }

      setFileQueue(selectedFiles.map(f => ({ file: f, status: 'pending' })));
    }
  };

  const startConversion = () => {
    if (fileQueue.length === 0) return;
    setIsConverting(true);
    setCancelRequested(false);
    setCurrentIndex(0);
  };

  const cancelConversion = () => {
    setCancelRequested(true);
    setIsConverting(false);
    setCurrentIndex(-1);
    
    // Reset pending files
    setFileQueue(prev => prev.map(f => 
      f.status === 'converting' || f.status === 'pending' 
        ? { ...f, status: 'pending' } 
        : f
    ));
  };

  const processNextFile = async (index: number) => {
    if (index >= fileQueue.length || cancelRequested) {
      setIsConverting(false);
      return;
    }

    // Set status to converting
    setFileQueue(prev => {
      const updated = [...prev];
      updated[index].status = 'converting';
      return updated;
    });

    try {
      const currentFile = fileQueue[index].file;
      const text = await currentFile.text();
      setCurrentMarkdown(text);
      
      // Give React time to render the markdown completely in the hidden container
      // And highlight.js time to process
      setTimeout(() => {
        if (cancelRequested) return;
        

        const pdfName = currentFile.name.replace(/\\.md$/i, '.pdf');
        
        exportPDF('batch-preview-container', pdfName, (exporting) => {
          if (!exporting) {
            // PDF export finished
            setFileQueue(prev => {
              const updated = [...prev];
              updated[index].status = 'success';
              return updated;
            });
            
            // Add a small delay between downloads to prevent browser freezing
            setTimeout(() => {
              if (!cancelRequested) {
                setCurrentIndex(index + 1);
              }
            }, 800);
          }
        });
      }, 500); // 500ms delay for DOM updates and layout
      
    } catch (error) {
      console.error('File reading failed', error);
      setFileQueue(prev => {
        const updated = [...prev];
        updated[index].status = 'error';
        updated[index].errorMessage = 'Failed to read file';
        return updated;
      });
      
      // Move to next even if it errors
      setTimeout(() => {
        if (!cancelRequested) {
          setCurrentIndex(index + 1);
        }
      }, 500);
    }
  };

  useEffect(() => {
    if (isConverting && currentIndex >= 0 && !cancelRequested) {
      if (currentIndex < fileQueue.length) {
        processNextFile(currentIndex);
      } else {
        // all done
        setIsConverting(false);
        setCurrentIndex(-1);
      }
    }
  }, [currentIndex, isConverting, cancelRequested]);

  const completedCount = fileQueue.filter(f => f.status === 'success' || f.status === 'error').length;
  const progressPercent = fileQueue.length > 0 ? Math.round((completedCount / fileQueue.length) * 100) : 0;

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/60 backdrop-blur-sm p-4">
      <div className="bg-white dark:bg-gray-800 rounded-xl shadow-2xl w-full max-w-2xl overflow-hidden flex flex-col max-h-[90vh]">
        <div className="flex justify-between items-center p-4 border-b border-gray-200 dark:border-gray-700 bg-gray-50 dark:bg-gray-900/50">
          <h2 className="text-lg font-semibold text-gray-900 dark:text-white flex items-center gap-2">
            <Type size={18} className="text-indigo-600 dark:text-indigo-400" />
            Convert Multiple .md to .pdf
          </h2>
          <button 
            onClick={onClose}
            disabled={isConverting}
            className="text-gray-500 hover:bg-gray-200 dark:hover:bg-gray-700 p-1.5 rounded-lg transition-colors disabled:opacity-50"
          >
            <X size={20} />
          </button>
        </div>

        <div className="p-6 flex-1 overflow-y-auto">
          {fileQueue.length === 0 ? (
            <div className="border-2 border-dashed border-gray-300 dark:border-gray-600 rounded-xl p-10 text-center flex flex-col items-center">
              <FileText size={48} className="text-gray-400 mb-4" />
              <p className="text-gray-600 dark:text-gray-300 mb-2">Select up to 20 Markdown files to convert</p>
              <input 
                type="file" 
                ref={fileInputRef} 
                onChange={handleFilesSelected} 
                accept=".md" 
                multiple 
                className="hidden" 
              />
              <button 
                onClick={() => fileInputRef.current?.click()}
                className="mt-4 px-4 py-2 bg-indigo-600 hover:bg-indigo-700 text-white rounded-lg font-medium transition-colors"
              >
                Browse Files
              </button>
            </div>
          ) : (
            <div className="space-y-4">
              <div className="flex justify-between items-center">
                <span className="font-medium text-gray-700 dark:text-gray-300">
                  {fileQueue.length} {fileQueue.length === 1 ? 'file' : 'files'} selected
                </span>
                {!isConverting && fileQueue.some(f => f.status === 'pending') && (
                  <button 
                    onClick={() => setFileQueue([])}
                    className="text-sm text-red-500 hover:underline"
                  >
                    Clear All
                  </button>
                )}
              </div>

              {/* Progress Bar */}
              {isConverting && (
                <div className="space-y-2 mb-6">
                  <div className="flex justify-between text-sm">
                    <span className="text-indigo-600 dark:text-indigo-400 font-medium">Converting {currentIndex + 1} of {fileQueue.length}...</span>
                    <span className="text-gray-500">{progressPercent}%</span>
                  </div>
                  <div className="w-full bg-gray-200 dark:bg-gray-700 rounded-full h-2.5 overflow-hidden">
                    <div className="bg-indigo-600 h-2.5 rounded-full transition-all duration-300" style={{ width: `${progressPercent}%` }}></div>
                  </div>
                </div>
              )}

              {/* File List */}
              <div className="border border-gray-200 dark:border-gray-700 rounded-lg max-h-64 overflow-y-auto bg-gray-50 dark:bg-gray-900/40">
                <ul className="divide-y divide-gray-200 dark:divide-gray-700">
                  {fileQueue.map((item, idx) => (
                    <li key={idx} className="p-3 flex items-center justify-between hover:bg-gray-100 dark:hover:bg-gray-800 transition-colors">
                      <div className="flex items-center gap-3 truncate pr-4">
                        <FileText size={16} className="text-gray-400 shrink-0" />
                        <span className="text-sm font-medium text-gray-700 dark:text-gray-300 truncate">
                          {item.file.name}
                        </span>
                      </div>
                      
                      <div className="shrink-0 flex items-center">
                        {item.status === 'pending' && <span className="text-xs text-gray-400">Waiting...</span>}
                        {item.status === 'converting' && <Loader size={16} className="text-indigo-500 animate-spin" />}
                        {item.status === 'success' && <CheckCircle size={18} className="text-green-500" />}
                        {item.status === 'error' && (
                          <div className="flex items-center gap-1 text-red-500" title={item.errorMessage}>
                            <AlertCircle size={16} />
                            <span className="text-xs">Failed</span>
                          </div>
                        )}
                      </div>
                    </li>
                  ))}
                </ul>
              </div>
            </div>
          )}
        </div>

        {fileQueue.length > 0 && (
          <div className="p-4 border-t border-gray-200 dark:border-gray-700 bg-gray-50 dark:bg-gray-900/50 flex justify-end gap-3">
            {isConverting ? (
              <button 
                onClick={cancelConversion}
                className="px-4 py-2 bg-red-100 dark:bg-red-900/30 text-red-600 dark:text-red-400 hover:bg-red-200 dark:hover:bg-red-800/40 rounded-lg font-medium transition-colors focus:ring-2 focus:ring-red-500 focus:outline-none"
              >
                Cancel Conversion
              </button>
            ) : fileQueue.some(f => f.status === 'pending') ? (
              <button 
                onClick={startConversion}
                className="px-6 py-2 bg-indigo-600 hover:bg-indigo-700 text-white rounded-lg font-medium transition-colors shadow-sm focus:ring-2 focus:ring-indigo-500 focus:outline-none"
              >
                Start Conversion
              </button>
            ) : (
              <button 
                onClick={onClose}
                className="px-6 py-2 bg-gray-200 dark:bg-gray-700 hover:bg-gray-300 dark:hover:bg-gray-600 text-gray-800 dark:text-gray-200 rounded-lg font-medium transition-colors"
              >
                Done
              </button>
            )}
          </div>
        )}
      </div>

      {/* Hidden container for rendering markdown elements for the PDF generator */}
      <div 
        aria-hidden="true" 
        style={{ position: 'absolute', top: '-9999px', left: '-9999px', width: '210mm' }}
      >
        <div id="batch-preview-container" className="markdown-body max-w-4xl mx-auto print-container" style={{ minHeight: 'max-content', padding: '20px' }}>
            <ReactMarkdown
              remarkPlugins={[remarkGfm]}
              rehypePlugins={[rehypeHighlight]}
            >
              {currentMarkdown}
            </ReactMarkdown>
        </div>
      </div>
    </div>
  );
};

export default BatchConverter;
